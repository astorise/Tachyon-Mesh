# Design: Air-Gapped Asset Data Model

## 1. The GitOps YAML Specification (Asset Bundle)
This file defines the strict cryptographic expectations for the payloads the node will execute.

    api_version: assets.tachyon.io/v1alpha1
    kind: AssetBundle
    metadata:
      name: core-services-bundle
      environment: production
      
    spec:
      artifacts:
        - name: "system-faas-authz"
          type: wasm_component
          sha256: "e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855"
          signature: "ey..." # Ed25519 signature generated by Tachyon-UI
          
        - name: "llama-3-8b-instruct"
          type: ai_model_gguf
          sha256: "8d4e9c7a2b5f..."
          
      # Global verification policy
      security:
        strict_signature_verification: true
        allowed_public_keys:
          - "ssh-ed25519 AAAAC3NzaC1lZDI1NTE5AAAAI..." # Trusted UI/Admin key

## 2. The WIT Contract (`wit/config-assets.wit`)
The strict Wasm interface used by `system-faas-config-api` to register these metadata expectations before receiving the physical byte stream.

    interface config-assets {
        enum asset-type { wasm-component, ai-model-gguf, static-blob }

        record artifact-metadata {
            name: string,
            asset-type: asset-type,
            sha256-hash: string,
            signature: option<string>,
        }

        record security-policy {
            strict-signature-verification: bool,
            allowed-public-keys: list<string>,
        }

        record asset-bundle {
            name: string,
            artifacts: list<artifact-metadata>,
            security: security-policy,
        }

        record asset-configuration {
            bundles: list<asset-bundle>,
        }

        /// Validation (Ensures hashes are valid SHA256 length, etc.)
        validate-asset-config: func(config: asset-configuration) -> result<_, string>;

        /// Metadata CRUD
        get-asset-config: func() -> result<asset-configuration, string>;
        apply-asset-bundle: func(bundle: asset-bundle) -> result<_, string>;
    }
